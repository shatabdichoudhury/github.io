64         public void paintShapeOnCanvas() {
65     
66             ShapeShadingType shapeShadingType = geometricShape.getShadingType();
67             Color primaryColor = geometricShape.getPrimaryColor();
68             Color secondaryColor = geometricShape.getSecondaryColor();
69     
70             IPaintable drawableShape = new BasicShapeStrategy();
71     
72             //Draw the shapes on the paintCanvas
73             switch (shapeShadingType) {
74                 case OUTLINE_AND_FILLED_IN:
75                     drawableShape = DrawableShapeFactory.outlineAndFilledShade(primaryColor, secondaryColor, shape, graphics2d);
76                     break;
77                 case FILLED_IN:
78                     drawableShape = DrawableShapeFactory.filledShade(primaryColor, shape, graphics2d);
79                     break;
80                 case OUTLINE:
81                     drawableShape = DrawableShapeFactory.outlineShade(primaryColor, shape, graphics2d);
82                     break;
83             }
84     
85             drawableShape.paintShape();
86         }

113          */
114         @Override
115         public IShape pasteShape() {
116     
117             DrawShape pastedShape = new DrawShape(this, paintCanvas);
118     
119             AffineTransform transform = new AffineTransform();
120             
121             // get copied shape position
122             
123             Integer xCoordCopiedShape = pastedShape.getShapeProperty().getShapeXcoord();
124             Integer width = pastedShape.getShapeProperty().getWidth();
125             Integer yCoordCopiedShape = pastedShape.getShapeProperty().getShapeYcoord();
126             Integer height = pastedShape.getShapeProperty().getHeight();
127            
128             //** paste the copied shape next to the original
129             
130             int posX =   xCoordCopiedShape + width + 1;
131             int posY =    yCoordCopiedShape + 1;
132             
133             // reset posX and posY if they go beyond screen
134             if ((posX > 900) || (posY > 300))
135             {
136             	posX = 1;
137             	posY = 1; 
138             }    
139             // the paste positions are still buggy
140             transform.translate(posX,posY);
141     
142             Shape offsetCopiedShape = transform.createTransformedShape(this.getShape());
143     
144             pastedShape.setShape(offsetCopiedShape);
145     
146             return pastedShape;
147         }
