21     	public ConnectEvent(IUiModule uiModule, IDrawCanvas paintCanvas, PaintObservable paintObservable) {

27     	@Override

32     	private void setupEvents() {
33     		uiModule.addEvent(EventName.UNDO, () -> processUndo());
34     		uiModule.addEvent(EventName.REDO, () -> processRedo());
35     		uiModule.addEvent(EventName.COPY, () -> processCopy());
36     		uiModule.addEvent(EventName.PASTE, () -> processPaste());
37     		uiModule.addEvent(EventName.DELETE, () -> processDelete());
38     		uiModule.addEvent(EventName.GROUP, () -> processGroup());
39     		uiModule.addEvent(EventName.UNGROUP, () -> processUngroup());
40     	}
41     
42     	public void processUndo() {

46     	}

57     		copyShapeCommand.run();
58     	}
59     
60     	public void processPaste() {

63     		pasteShapeCommand.run();
64     	}
65     
66     	public void processDelete() {

69     		deleteShapeCommand.run();
70     	}
71

75     		groupShapeCommand.run();
76     	}
77     
78     	private void processUngroup() {
79     		UngroupShape ungroupShape = new UngroupShape(paintCanvas);
