12     public class ShapeFactory {
13     
14     	public static Shape createShape(GeometricShape geometricShape) {
15     		ShapeType shapeType = geometricShape.getShapeType();
16     		Integer shapeXcoord = geometricShape.getShapeXcoord();
17     		Integer shapeYcoord = geometricShape.getShapeYcoord();
18     		Integer width = geometricShape.getWidth();
19     		Integer height = geometricShape.getHeight();
20     		Point startPoint = geometricShape.getStartPoint();
21     		Point endPoint = geometricShape.getEndPoint();
22     
23     		// find the shape type
24     		IShapeTypeStrategy shapeTypeStrategy = null;
25     
26     		if (shapeType.equals(ShapeType.RECTANGLE)) {
27     			shapeTypeStrategy = new Rectangle(shapeXcoord, shapeYcoord, width, height);
28     		} else if (shapeType.equals(ShapeType.ELLIPSE)) {
29     			shapeTypeStrategy = new Ellipse(shapeXcoord, shapeYcoord, width, height);
30     		} else if (shapeType.equals(ShapeType.TRIANGLE)) {
31     			shapeTypeStrategy = new Triangle(startPoint, endPoint);
32     		}
33     		return shapeTypeStrategy.create();
34     	}
35     }
