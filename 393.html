1     package view;
2     
3     import view.Enum.EventName;
4     import view.interfaces.IGuiControl;
5     
6     import javax.swing.*;
7     import javax.swing.border.Border;
8     import javax.swing.border.CompoundBorder;
9     import javax.swing.border.EmptyBorder;
10     import javax.swing.border.LineBorder;
11     import java.awt.*;
12     import java.util.HashMap;
13     import java.util.Map;
14     import java.util.NoSuchElementException;
15     
16     public class GuiWindow extends JFrame implements IGuiControl {
17     
18     	private static final long serialVersionUID = 1L;
19     	private static final int defaultWidth = 1250;
20     	private static final int defaultHeight = 800;
21     	private static final String defaultTitle = "Shatabdi Choudhury JPaint";
22     
23     	private final Insets defaultButtonDimensions = new Insets(5, 8, 5, 8);
24     	private final Map<EventName, JButton> eventButtons = new HashMap<>();
25     
26     	public GuiWindow(JComponent canvas) {
27     		setVisible(true);
28     		setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
29     		setTitle(defaultTitle);
30     		setSize(defaultWidth, defaultHeight);
31     		setExtendedState(JFrame.MAXIMIZED_BOTH);
32     		JPanel window = createWindow();
33     		window.add(canvas, BorderLayout.CENTER);
34     		validate();
35     
36     		System.out.println("Gui window appears.");
37     	}
38     
39     	@Override
40     	public JButton getButton(EventName eventName) {
41     		if (!eventButtons.containsKey(eventName))
42     			throw new NoSuchElementException("Button does not exists for action " + eventName.toString());
43     
44     		return eventButtons.get(eventName);
45     	}
46     
47     	private JPanel createWindow() {
48     		JPanel contentPane = createBackgroundPanel();
49     		JPanel buttonPanel = createMenu();
50     		contentPane.add(buttonPanel, BorderLayout.NORTH);
51     		return contentPane;
52     	}
53     
54     	private JPanel createMenu() {
55     		JPanel buttonPanel = createButtonPanel();
56     
57     		for (EventName eventName : EventName.values()) {
58     			addButtonToPanel(eventName, buttonPanel);
59     		}
60     
61     		return buttonPanel;
62     	}
63     
64     	private void addButtonToPanel(EventName eventName, JPanel panel) {
65     		JButton newButton = createButton(eventName);
66     		eventButtons.put(eventName, newButton);
67     		panel.add(newButton);
68     	}
69     
70     	private JButton createButton(EventName eventName) {
71     		JButton newButton = new JButton(eventName.toString());
72     		newButton.setForeground(Color.BLACK);
73     		newButton.setBackground(Color.WHITE);
74     		newButton.setBorder(createButtonBorder());
75     		return newButton;
76     	}
77     
78     	private Border createButtonBorder() {
79     		Border line = new LineBorder(Color.BLACK);
80     		Border margin = new EmptyBorder(defaultButtonDimensions);
81     		return new CompoundBorder(line, margin);
82     	}
83     
84     	private JPanel createButtonPanel() {
85     		JPanel panel = new JPanel();
86     		FlowLayout flowLayout = (FlowLayout) panel.getLayout();
87     		flowLayout.setAlignment(FlowLayout.LEFT);
88     		panel.setBackground(new Color(156, 156, 156));
89     		return panel;
90     	}
91     
92     	private JPanel createBackgroundPanel() {
93     		JPanel contentPane = new JPanel();
94     		contentPane.setBorder(new EmptyBorder(0, 0, 0, 0));
95     		contentPane.setLayout(new BorderLayout(0, 0));
96     		contentPane.setBackground(Color.WHITE);
97     		setContentPane(contentPane);
98     		return contentPane;
99     	}
100     }
