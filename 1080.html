26     	@Override
27         public <T> T getDialogResponse(IDialog dialogSettings) {
28             Object selectedValue = JOptionPane.showInputDialog(null,
29                     dialogSettings.getDialogText(), dialogSettings.getDialogTitle(),
30                     JOptionPane.PLAIN_MESSAGE,
31                     null,
32                     dialogSettings.getDialogOptions(),
33                     dialogSettings.getCurrentSelection());
34             return selectedValue == null
35                     ? (T) dialogSettings.getCurrentSelection()
36                     : (T) selectedValue;
37         }
